### Features

Chaos engineering module for SimpleW to simulate HTTP errors, latency, and connection failures.

### Getting Started

The minimal API

```cs
using SimpleW;
using SimpleW.Service.Chaos;

namespace Sample {
    class Program {

        static async Task Main() {

            // listen to all IPs port 2015
            var server = new SimpleWServer(IPAddress.Any, 2015);

            server.MapGet("/api/test", () => {
                return new { message = "Hello World !" };
            });

            server.UseChaosModule(options => {
                options.Enabled = true;
                options.Prefix = "/api";
                options.Probability = 0.10; // 10% of requests fail
            });

            server.OnStarted(s => {
                Console.WriteLine("server started at http://localhost:{server.Port}/");
            });

            // start a blocking background server
            await server.RunAsync();

        }
    }

}
```
